import subprocess
import multiprocessing
import time

def optimize_n(n, hours=8):
    """Run bbox3 on a single N value for extended time."""
    cmd = [
        './bbox3',
        '-i', f'input_n{n}.csv',
        '-o', f'output_n{n}.csv',
        '-n', '100000',  # iterations
        '-r', '500',     # restarts
        '-t', str(int(hours * 3600))  # time limit in seconds
    ]
    subprocess.run(cmd)

# Run in parallel on all N values
