    # Variables: position (x, y) for each tree (discretized)
    x = [solver.IntVar(0, grid_size, f'x_{i}') for i in range(n_trees)]
    y = [solver.IntVar(0, grid_size, f'y_{i}') for i in range(n_trees)]
    
    # Bounding box variables
    max_coord = solver.IntVar(0, grid_size, 'max_coord')
    
    # Constraints: max tracking
    for i in range(n_trees):
        solver.Add(max_coord >= x[i])
        solver.Add(max_coord >= y[i])
    
    # Non-overlap constraints (using big-M formulation)

    # Objective: minimize bounding box
