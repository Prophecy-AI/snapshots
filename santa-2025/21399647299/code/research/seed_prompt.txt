# Asymmetric tessellation approach
def create_asymmetric_tessellation(n, angle1=67, angle2=247, spacing_x=0.5, spacing_y=0.5):
    """
    Create a tessellation with alternating tree orientations.
    Trees at angle1 and angle2 (180° apart) interlock better than uniform angles.
    """
    trees = []
    side = int(math.ceil(math.sqrt(n)))
    
    for row in range(side):
        for col in range(side):
            if len(trees) >= n:
                break
            
            x = col * spacing_x
            y = row * spacing_y
            
            # Alternate angles in checkerboard pattern
            if (row + col) % 2 == 0:
                angle = angle1
            else:
                angle = angle2
            
            trees.append((x, y, angle))
    
    return trees[:n]
